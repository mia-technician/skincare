// 创建多层级下拉菜单（品牌 → 产品） function createSearchableSelect(timeOfDay, containerElement) { const wrapper = document.createElement('div'); wrapper.className = 'cascade-select'; const input = document.createElement('input'); input.type = 'text'; input.className = 'main-input'; input.placeholder = timeOfDay === 'morning' ? '🔍 选择品牌...' : '🔍 选择品牌...'; input.autocomplete = 'off'; input.readOnly = true; const panel = document.createElement('div'); panel.className = 'cascade-panel'; wrapper.appendChild(input); wrapper.appendChild(panel); containerElement.appendChild(wrapper); let currentLevel = 'brand'; // 'brand' 或 'product' let currentBrand = ''; let allBrands = []; function rebuildBrandList() { // 获取所有品牌并排序 const brandsSet = new Set(); productsData.forEach(p => { if (p.brand) brandsSet.add(p.brand); }); allBrands = Array.from(brandsSet).sort((a, b) => a.localeCompare(b)); let html = `
`; for (const brand of allBrands) { const productCount = productsData.filter(p => p.brand === brand).length; html += `
📁 ${escapeHtml(brand)} ▶ ${productCount}
`; } html += `
`; panel.innerHTML = html; // 绑定品牌点击事件 panel.querySelectorAll('.cascade-brand-item').forEach(item => { item.onclick = (e) => { e.stopPropagation(); currentBrand = item.dataset.brand; currentLevel = 'product'; input.value = currentBrand; showProductsByBrand(currentBrand); }; }); } function showProductsByBrand(brand) { const brandProducts = productsData .filter(p => p.brand === brand) .sort((a, b) => a.name.localeCompare(b.name)); let html = `
← 返回品牌列表
`; for (const p of brandProducts) { html += `
${escapeHtml(p.name)} $${p.price}
`; } html += `
`; panel.innerHTML = html; // 绑定返回按钮 const backBtn = panel.querySelector('.cascade-back-btn'); if (backBtn) { backBtn.onclick = () => { currentLevel = 'brand'; input.value = ''; rebuildBrandList(); }; } // 绑定产品点击事件 panel.querySelectorAll('.cascade-product-item').forEach(item => { item.onclick = (e) => { e.stopPropagation(); const productId = item.dataset.productId; const product = getProductById(productId); if (product) { if (timeOfDay === 'morning') { currentMorningProducts.push({ productId: product.id, time: 'morning' }); } else { currentEveningProducts.push({ productId: product.id, time: 'evening' }); } renderRoutines(); updateTotalPrice(); saveToLocalStorage(); } input.value = ''; panel.classList.remove('show'); currentLevel = 'brand'; rebuildBrandList(); }; }); } // 点击输入框显示品牌列表 input.onclick = () => { if (currentLevel === 'brand') { rebuildBrandList(); } else if (currentLevel === 'product') { showProductsByBrand(currentBrand); } panel.classList.add('show'); }; // 点击其他地方关闭 document.addEventListener('click', function(e) { if (!wrapper.contains(e.target)) { panel.classList.remove('show'); } }); // 监听产品库更新 const observer = new MutationObserver(() => { if (currentLevel === 'brand') { rebuildBrandList(); } }); observer.observe(document.getElementById('brandProductsContainer') || document.body, { childList: true, subtree: true }); // 初始化 rebuildBrandList(); }
