        // Helper Functions

        function getProductById(id) { return productsData.find(p => p.id === id); }

        function generateId() { return Date.now().toString(36) + Math.random().toString(36).substr(2, 6); }

        function escapeHtml(str) { if (!str) return ''; return str.replace(/[&<>]/g, function(m) { if(m==='&') return '&amp;'; if(m==='<') return '&lt;'; if(m==='>') return '&gt;'; return m; }); }

        

        function loadRoutine(routineId) {

            let routine = savedRoutines[routineId];

            if (routine) {

                currentRoutineId = routineId;

                morningProducts = routine.morning.map(pid => ({ productId: pid, time: "morning" }));

                eveningProducts = routine.evening.map(pid => ({ productId: pid, time: "evening" }));

                renderRoutines();

                updateTotalPrice();

                highlightActiveRoutine();

            }

        }

        

        function highlightActiveRoutine() {

            document.querySelectorAll('.routine-btn').forEach(btn => {

                if (btn.dataset.id === currentRoutineId) btn.classList.add('active');

                else btn.classList.remove('active');

            });

        }

        

        function saveCurrentAsStandardRoutine() {

            let name = prompt("Enter a name for this new Standard Routine:", "My Custom Routine");

            if (!name || name.trim() === "") return;

            let newId = "standard-" + Date.now();

            savedRoutines[newId] = {

                name: name.trim(),

                morning: morningProducts.map(p => p.productId),

                evening: eveningProducts.map(p => p.productId)

            };

            renderRoutineButtons();

            loadRoutine(newId);

            alert(`"${name}" has been saved as a Standard Routine!`);

        }

        }

        function deleteCurrentRoutine() {

        function getUniqueProductsTotal() {

            let ids = new Set();

            morningProducts.forEach(i => ids.add(i.productId));

            eveningProducts.forEach(i => ids.add(i.productId));

            let total = 0;

            ids.forEach(id => { let p = getProductById(id); if(p) total += p.price; });

            return total;

        }

        function attachDragEvents() {

            document.querySelectorAll('.library-item').forEach(el => {

                el.setAttribute('draggable', 'true');

                el.addEventListener('dragstart', (e) => {

                        price: parseFloat(el.dataset.productPrice), benefit: el.dataset.productBenefit,

                        usage: el.dataset.productUsage, image: el.dataset.productImage

                });

            });

        }

        function saveProduct() {

            let name = document.getElementById('productName').value.trim();

            if (!name) { alert("Enter product name"); return; }

            let newProduct = {

                id: document.getElementById('editProductId').value || generateId(),

                name: name,

                brand: document.getElementById('productBrand').value,

                price: parseFloat(document.getElementById('productPrice').value) || 0,

                benefit: document.getElementById('productBenefit').value,

                usage: document.getElementById('productUsage').value,

                image: document.getElementById('productImage').value

            };

            let existing = productsData.findIndex(p => p.id === newProduct.id);

            if (existing >= 0) productsData[existing] = newProduct;

            else productsData.push(newProduct);

            closeForm();

            renderBrandProducts();

            renderRoutines();

            updateTotalPrice();

        }

        

        function closeForm() {

            document.getElementById('productForm').classList.remove('show');

            document.getElementById('toggleAddBtn').style.display = '';

        }

        

        function renderRoutines() {

            let morningContainer = document.getElementById('morningList');

            let eveningContainer = document.getElementById('eveningList');

            if (morningContainer) morningContainer.innerHTML = morningProducts.map((item, idx) => renderCard(item.productId, idx, 'morning')).join('');

            if (eveningContainer) eveningContainer.innerHTML = eveningProducts.map((item, idx) => renderCard(item.productId, idx, 'evening')).join('');

            attachDropEvents('morningList', 'morning');

            attachDropEvents('eveningList', 'evening');

            attachDeleteEvents();

        }

        

        function renderCard(pid, idx, time) {

            let p = getProductById(pid);

            if (!p) return '';

            return `<div class="product-card" data-product-id="${p.id}" data-index="${idx}" data-time="${time}" draggable="true">

                <button class="delete-btn" data-time="${time}" data-index="${idx}">✕</button>

                <div class="product-image"><img src="${p.image || 'https://via.placeholder.com/52/f1e8e0/8f7e6b?text=Product'}" alt="${escapeHtml(p.name)}"></div>

                <div class="product-info">

                    <div class="product-name">${escapeHtml(p.name)} <span class="product-price">$${p.price}</span></div>

                    <div class="benefit">${escapeHtml(p.benefit || '')}</div>

                    <div class="usage"><i class="icon-hand"></i> ${escapeHtml((p.usage || '').substring(0, 55))}</div>

                </div>

            </div>`;

        }

        

        function attachDropEvents(containerId, time) {

            let container = document.getElementById(containerId);

            if (!container) return;

            container.ondragover = (e) => e.preventDefault();

            container.ondrop = (e) => {

                e.preventDefault();

                let targetCard = e.target.closest('.product-card');

                if (!targetCard) return;

                let targetIdx = parseInt(targetCard.dataset.index);

                let dragData = JSON.parse(e.dataTransfer.getData('text/plain'));

                if (dragData && dragData.id) {

                    if (time === 'morning' && morningProducts[targetIdx]) morningProducts[targetIdx] = { productId: dragData.id, time: 'morning' };

                    if (time === 'evening' && eveningProducts[targetIdx]) eveningProducts[targetIdx] = { productId: dragData.id, time: 'evening' };

                    updateCurrentRoutine();

                    renderRoutines();

                    updateTotalPrice();

                }

            };

        }

        

        function attachDeleteEvents() {

            document.querySelectorAll('.delete-btn').forEach(btn => {

                btn.onclick = (e) => {

                    e.stopPropagation();

                    let time = btn.dataset.time, idx = parseInt(btn.dataset.index);

                    if (time === 'morning') morningProducts.splice(idx, 1);

                    else eveningProducts.splice(idx, 1);

                    updateCurrentRoutine();

                    renderRoutines();

                    updateTotalPrice();

                };

            });

        }
